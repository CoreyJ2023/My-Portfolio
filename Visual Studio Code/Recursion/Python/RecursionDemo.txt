Module Main()
	// By passing the argument 5 to the message module,
	// we are telling it to display the message
	// five times.
	Call message(5)
End Module

Module Message(Integer times)
	If times > 0 Then
		Display "this is a recursive module."
		Call message(times - 1)
	End If
End Module
